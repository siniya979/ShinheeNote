private Tuple2<File, byte[]> getTempFileFromBase64(String base64) {  
    Tuple2<String, byte[]> fileInfo = Base64ImgConverter.convertToImage(base64);  
    return Tuple.of(tempStorage.save(fileInfo._2, "GC", fileInfo._1()), fileInfo._2());  
}
